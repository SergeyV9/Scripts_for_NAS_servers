# Проверяем на наличие файлов в каталоге
if find -name '*.pdf' ; then

# Создаем переменную с текущей датой в формате год_месяц_день
DIRDATE=`/bin/date '+%y_%m_%d'`

# Создание каталога с именем "год_месяц_день"
mkdir "$DIRDATE"

# Перемещение файлов по расширению
mv *.PDF *.pdf "$DIRDATE"

# Архивация (7z) созданного каталога
export LANG=ru_RU.UTF-8
7z a -mx5 -m0=lzma2 "$DIRDATE".7z "$DIRDATE"

# Удаление заархивированного каталога
rm -rf "$DIRDATE"

fi
